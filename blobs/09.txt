    const stage = RPG.STAGES[stageIndex];
    const rng = RPG.makeRng(profile.id + ':' + stage.id + ':' + profile.counters.stages);
    const w = {
      stageIndex, stage, rng, width: stage.width, elapsed: 0, remaining: stage.timeLimit,
      cameraX: 0, shake: 0, objectiveCount: 0, objectiveDone: false,
      keyFound: false, gateOpen: false, cold: 0, lavaWave: 0, lavaWarning: 0,
      kills: 0, chestsOpened: 0, elitesKilled: 0, deaths: 0, bossKilled: false,
      enemies: [], projectiles: [], enemyProjectiles: [], effects: [], floats: [], drops: [],
      platforms: [], gaps: [], chests: [], braziers: [], cores: [],
      gate: { x: stage.width - 145, y: GROUND_Y - 92, w: 64, h: 92 },
      checkpoint: 70, finished: false, phoenixUsed: false,
    };

    for (let x = 520; x < stage.width - 700; x += 680 + rng.int(-100, 120)) {
      if (rng.random() < 0.55) {
        const width = rng.int(130, 260), y = GROUND_Y - rng.int(90, 160);
        w.platforms.push({ x, y, w: width, h: 18, kind: 'stone' });
      }
    }
    for (let x = 1050; x < stage.width - 900; x += 1250 + rng.int(-160, 180)) {
      const width = rng.int(110, 170);
      w.gaps.push([x, x + width]);
      w.platforms.push({ x: x - 25, y: GROUND_Y - 80, w: width + 50, h: 16, kind: 'bridge' });
    }

    const enemyCount = 10 + stageIndex * 3;
    for (let i = 0; i < enemyCount; i++) {
      const x = 430 + i * ((stage.width - 1050) / enemyCount) + rng.int(-80, 90);
      const type = rng.pick(stage.pools);
      let eliteId = null;
      if (rng.random() < 0.08 + stageIndex * 0.025) eliteId = rng.pick(Object.keys(RPG.ELITE_AFFIXES));
      w.enemies.push(makeEnemy(type, x, stageIndex, eliteId, rng));
    }
    if (stage.mechanic === 'eliteHunt' && !w.enemies.some(e => e.eliteId)) {
      const target = w.enemies[Math.floor(w.enemies.length * 0.65)];
      const afId = rng.pick(Object.keys(RPG.ELITE_AFFIXES));
      const replacement = makeEnemy(target.type, target.x, stageIndex, afId, rng);
      w.enemies[w.enemies.indexOf(target)] = replacement;
    }
    if (stage.mechanic === 'boss') {
      w.enemies.push(makeEnemy('boss', stage.width - 620, stageIndex, 'fortified', rng));
    }

    const chestPositions = [Math.floor(stage.width * 0.27), Math.floor(stage.width * 0.55), Math.floor(stage.width * 0.78)];
    chestPositions.forEach((x, i) => w.chests.push({ x, y: GROUND_Y - 32, w: 36, h: 32, opened: false, key: stage.mechanic === 'keyDoor' && i === 1, rarity: i === 2 ? 'rare' : 'magic' }));
    if (stage.mechanic === 'cold') [850, 1950, 3150, 4300].forEach(x => w.braziers.push({ x, y: GROUND_Y - 38, w: 34, h: 38 }));
    if (stage.mechanic === 'lava') [1250, 2720, 4300].forEach(x => w.cores.push({ x, y: GROUND_Y - 62, w: 46, h: 62, hp: 160 + stage.level * 14, maxHp: 160 + stage.level * 14, dead: false }));
    return w;
  }

  function createPlayer() {
    stats = RPG.deriveStats(profile);
    return {
      x: 70, y: GROUND_Y - 46, w: 32, h: 46, vx: 0, vy: 0, face: 1,
      hp: stats.maxHp, mp: stats.maxMp, onGround: false, coyote: 0, jumpBuffer: 0,
      attackCd: 0, attackT: 0, combo: 0, comboT: 0,
      primaryCd: 0, secondaryCd: 0, invuln: 0, guardT: 0, dashT: 0,
      slowT: 0, freePrimary: false, potionLock: 0,
    };
  }

  function startStage(stageIndex) {
    if (!profile || stageIndex > profile.unlockedStage || !RPG.STAGES[stageIndex]) return false;
    stats = RPG.deriveStats(profile);
    world = buildWorld(stageIndex);
    player = createPlayer();
    setState(STATE.FIELD, null);
    try { Audio.musicStart(stageIndex % 3); } catch (_) {}
    toast(world.stage.objective, world.stage.theme.accent);
    return true;
  }

  function hasUnique(uniqueId) {
    return RPG.SLOTS.some(slot => profile.equipment[slot] && String(profile.equipment[slot].id).startsWith(uniqueId));
  }

  function addFloat(x, y, text, color, size) {
    if (!world) return;
    world.floats.push({ x, y, text: String(text), color: color || '#fff', size: size || 12, life: 1.1, maxLife: 1.1 });
    if (world.floats.length > 40) world.floats.shift();
  }

  function addEffect(type, x, y, options) {
    if (!world) return;
    world.effects.push(Object.assign({ type, x, y, life: 0.5, maxLife: 0.5, r: 12, color: '#fff' }, options || {}));
    if (world.effects.length > 90) world.effects.shift();
  }

  function enemyAtPoint(x, y, radius) {
    return world.enemies.filter(e => !e.dead && Math.hypot(centerX(e) - x, centerY(e) - y) <= radius);
  }

  function rollPlayerDamage(mult, skill, target) {
    const variance = 0.88 + Math.random() * 0.24;
    let damage = stats.attack * mult * variance;
    if (skill) damage *= stats.skillPower;
    if (target && target.hp / target.maxHp <= 0.3) damage *= 1 + (stats.effects.executePct || 0) / 100;
    if (target && (target.role === 'boss' || target.eliteId)) damage *= 1 + (stats.effects.bossDamagePct || 0) / 100;
    if (hasUnique('wintercrown') && target && target.frozen > 0) damage *= 1.3;
    if (hasUnique('archon') && player.mp >= stats.maxMp * 0.8 && skill) damage *= 1.2;
    const crit = Math.random() * 100 < stats.crit;
    if (crit) damage *= stats.critDamage / 100;
    return { damage: Math.max(1, Math.round(damage)), crit };
  }

  function damageCore(core, amount) {
    if (!core || core.dead) return;
    core.hp -= amount;
    addFloat(core.x + core.w / 2, core.y - 8, amount, '#ffb04f');
    addEffect('burst', core.x + core.w / 2, core.y + core.h / 2, { color: '#ff6b2c', r: 18 });
    if (core.hp <= 0) {
      core.dead = true;
      world.objectiveCount++;
      sfx('break');
      toast('균열 핵 파괴 ' + world.objectiveCount + '/' + world.stage.objectiveTarget, '#ff8c42');
      updateObjective();
    }
  }

  function damageEnemy(enemy, amount, options) {
    if (!enemy || enemy.dead) return false;
    const opts = options || {};
    let damage = Math.max(1, Math.round(amount));
    const incomingFromFront = Math.sign(centerX(player) - centerX(enemy)) === enemy.face;
    if (enemy.role === 'shield' && incomingFromFront && !opts.breakGuard) {
      damage = Math.max(1, Math.round(damage * 0.28));
      addFloat(centerX(enemy), enemy.y - 6, '방어', '#8ed4ff');
    }
    enemy.hp -= damage;
    enemy.hitFlash = 0.12;
    enemy.aggro = true;
    enemy.vx += (opts.dir == null ? player.face : opts.dir) * (opts.knockback || 1.5);
    if (opts.freeze) enemy.frozen = Math.max(enemy.frozen, opts.freeze);
    if (opts.burn) { enemy.burn = Math.max(enemy.burn, opts.burn.duration); enemy.burnDps = Math.max(enemy.burnDps, opts.burn.dps); }
    addFloat(centerX(enemy), enemy.y - 8, damage + (opts.crit ? '!' : ''), opts.crit ? '#ffe66d' : (opts.color || '#fff'), opts.crit ? 17 : 12);
    addEffect('hit', centerX(enemy), centerY(enemy), { color: opts.color || '#fff', r: opts.crit ? 18 : 12, life: 0.22, maxLife: 0.22 });
    if (stats.leech > 0) player.hp = Math.min(stats.maxHp, player.hp + damage * stats.leech / 100);
    if (enemy.hp <= 0) killEnemy(enemy, opts);
    return enemy.dead;
  }

  function spawnDrop(x, y, kind, payload) {
    world.drops.push({ x, y, w: 24, h: 24, kind, payload, vy: -4 - Math.random() * 2, vx: (Math.random() * 2 - 1) * 2.2, life: 45, picked: false });
  }

  function killEnemy(enemy, options) {
    if (enemy.dead) return;
    enemy.dead = true;
    world.kills++;
    const opts = options || {};
    if (enemy.eliteId) world.elitesKilled++;
    if (enemy.role === 'boss') { world.bossKilled = true; world.objectiveCount = 1; }
    const levelUps = RPG.awardXP(profile, enemy.def.xp + world.stage.level * 2);
    if (levelUps) { stats = RPG.deriveStats(profile); player.hp = stats.maxHp; player.mp = stats.maxMp; toast('레벨 업! 스킬 포인트와 능력치 포인트를 획득했습니다.', '#ffe66d'); sfx('levelup'); }
    const gold = enemy.def.gold[0] + Math.floor(Math.random() * (enemy.def.gold[1] - enemy.def.gold[0] + 1));
    spawnDrop(centerX(enemy), enemy.y, 'gold', gold + world.stage.level);
    let itemChance = enemy.role === 'boss' ? 1 : enemy.eliteId ? 0.72 : 0.14;
    if (Math.random() < itemChance) {
      const rarity = enemy.role === 'boss' ? (Math.random() < 0.35 ? 'unique' : 'epic') : enemy.eliteId ? 'rare' : undefined;
      spawnDrop(centerX(enemy) + 12, enemy.y, 'item', RPG.createItem(world.stage.level + (enemy.eliteId ? 2 : 0), world.rng, { rarity, luck: enemy.eliteId ? 12 : 0 }));
    }
    if (Math.random() < 0.13) spawnDrop(centerX(enemy) - 10, enemy.y, Math.random() < 0.72 ? 'hp' : 'mp', 1);
    if (enemy.elite && enemy.eliteId === 'explosive') {
      addEffect('explosion', centerX(enemy), centerY(enemy), { color: '#ff8b35', r: 90, life: 0.6, maxLife: 0.6 });
      if (Math.hypot(centerX(player) - centerX(enemy), centerY(player) - centerY(enemy)) < 90) hurtPlayer(enemy.damage * 0.8, enemy);
    }
    const combustChance = stats.effects.combustChance || 0;
    if (combustChance && (opts.burn || enemy.burn > 0) && Math.random() * 100 < combustChance) {
      addEffect('explosion', centerX(enemy), centerY(enemy), { color: '#ff6b35', r: 82, life: 0.52, maxLife: 0.52 });
      for (const nearby of enemyAtPoint(centerX(enemy), centerY(enemy), 82)) {
        const roll = rollPlayerDamage(0.65, true, nearby);
        damageEnemy(nearby, roll.damage, { crit: roll.crit, color: '#ff6b35', burn: { duration: 2.2, dps: Math.max(2, stats.fireDamage) } });
      }
    }
    if (hasUnique('bloodmoon') && opts.crit && Math.random() < 0.18) spawnDrop(centerX(enemy), enemy.y - 12, 'hp', 1);

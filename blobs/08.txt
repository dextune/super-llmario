  }

  function findForgeItem(key) {
    if (!key) return null;
    const [where, id] = key.split(':');
    if (where === 'eq') return profile.equipment[id] || null;
    if (where === 'bag') return profile.inventory[Number(id)] || null;
    return null;
  }

  function renderForge() {
    if (!profile) return;
    const wallet = $('forge-wallet');
    if (wallet) wallet.textContent = fmt(profile.gold) + 'G · 재료 ' + fmt(profile.materials);
    const all = [];
    for (const slot of RPG.SLOTS) if (profile.equipment[slot]) all.push({ key: 'eq:' + slot, item: profile.equipment[slot] });
    profile.inventory.forEach((item, i) => all.push({ key: 'bag:' + i, item }));
    if (!forgeSelection || !findForgeItem(forgeSelection)) forgeSelection = all.length ? all[0].key : null;
    const host = $('forge-items');
    if (host) host.innerHTML = all.map(entry => itemCard(entry.item, { action: 'forge-select', index: entry.key, selected: entry.key === forgeSelection })).join('');
    const detail = $('forge-detail');
    const item = findForgeItem(forgeSelection);
    if (detail) {
      if (!item) detail.innerHTML = '<p class="rpg-dim">강화할 장비가 없습니다.</p>';
      else {
        const up = RPG.upgradeCost(item), rr = RPG.rerollCost(item);
        detail.innerHTML = '<h3 style="color:' + RPG.RARITIES[item.rarity].color + '">' + esc(item.name) + (item.upgrade ? ' +' + item.upgrade : '') + '</h3>' +
          '<p>' + itemStatsHtml(item) + '</p>' +
          '<button data-action="forge-upgrade">강화 · ' + fmt(up.gold) + 'G / 재료 ' + up.materials + '</button>' +
          '<button data-action="forge-reroll" ' + ((!item.affixes || !item.affixes.length || item.rarity === 'unique') ? 'disabled' : '') + '>마지막 옵션 재련 · ' + fmt(rr.gold) + 'G / 재료 ' + rr.materials + '</button>';
      }
    }
  }

  function openForge() {

  function makeEnemy(type, x, stageIndex, eliteId, rng) {
    const def = RPG.ENEMIES[type];
    const stage = RPG.STAGES[stageIndex];
    const scale = 1 + stage.level * 0.17 + Math.max(0, profile.level - stage.level) * 0.025;
    const e = {
      id: 'enemy-' + rng.int(10000, 99999), type, def, role: def.role,
      x, y: type === 'bat' ? GROUND_Y - 150 : GROUND_Y - (type === 'boss' ? 86 : 42),
      w: type === 'boss' ? 82 : (type === 'charger' ? 48 : 38), h: type === 'boss' ? 86 : (type === 'bat' ? 30 : 42),
      vx: 0, vy: 0, face: -1,
      maxHp: Math.round(def.hp * scale), hp: Math.round(def.hp * scale),
      damage: Math.round(def.damage * (1 + stage.level * 0.1)),
      speed: def.speed, attackCd: rng.random() * 1.2, skillCd: rng.random() * 2,
      state: 'idle', stateT: 0, aggro: false, dead: false, hitFlash: 0,
      frozen: 0, burn: 0, burnDps: 0, eliteId: eliteId || null, elite: null,
      bossPhase: 1, summonCount: 0, blinkCd: 4 + rng.random() * 2,
    };
    if (eliteId) {
      const af = RPG.ELITE_AFFIXES[eliteId];
      e.elite = af;
      e.maxHp = Math.round(e.maxHp * af.hp); e.hp = e.maxHp;
      e.damage = Math.round(e.damage * af.damage); e.speed *= af.speed;
    }
    return e;
  }

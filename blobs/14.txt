    const k = clamp(e.life / e.maxLife, 0, 1);
    g.save(); g.globalAlpha = k * 0.8; g.strokeStyle = e.color; g.fillStyle = e.color;
    if (['shockwave', 'nova', 'aura', 'explosion'].includes(e.type)) {
      g.lineWidth = e.type === 'aura' ? 3 : 6; const r = e.r * (1 - k * 0.25); g.beginPath(); g.arc(e.x, e.y, r, 0, Math.PI * 2); g.stroke();
    } else if (e.type === 'slash') {
      g.lineWidth = 7; g.beginPath(); g.arc(e.x, e.y, e.r, -1.1, 1.1); g.stroke();
    } else if (e.type === 'dash') {
      g.fillRect(e.x - e.r * (e.face || 1), e.y - 16, e.r * (e.face || 1), 32);
    } else {
      g.beginPath(); g.arc(e.x, e.y, e.r * (1.2 - k * 0.2), 0, Math.PI * 2); g.fill();
    }
    g.restore();
  }

  function objectiveText() {
    if (!world) return '';
    const stage = world.stage;
    if (stage.objectiveType === 'survive') return '혹한 ' + Math.round(world.cold) + '% · 관문 도달';
    return stage.objective + ' [' + Math.min(world.objectiveCount, stage.objectiveTarget) + '/' + stage.objectiveTarget + ']';
  }

  function drawHUD() {
    if (!world || !player) return;
    g.fillStyle = 'rgba(5,9,18,.78)'; g.fillRect(0, 0, VIEW_W, 86);
    g.font = '9px "Press Start 2P", monospace'; g.textBaseline = 'top';
    g.fillStyle = '#ffe66d'; g.fillText('LV ' + profile.level + ' ' + RPG.CLASS_DEFS[profile.classId].name, 18, 10);
    drawBar(18, 27, 210, 14, player.hp / stats.maxHp, '#e94f52');
    g.fillStyle = '#fff'; g.fillText('HP ' + Math.ceil(player.hp) + '/' + stats.maxHp, 24, 30);
    drawBar(18, 47, 210, 12, player.mp / stats.maxMp, '#477cff');
    g.fillStyle = '#fff'; g.fillText('MP ' + Math.ceil(player.mp) + '/' + stats.maxMp, 24, 49);
    drawBar(18, 65, 210, 8, profile.xp / RPG.xpForLevel(profile.level), '#ffd45c');
    g.fillStyle = '#c9d5ec'; g.fillText(world.stage.name, 255, 10);
    g.fillStyle = world.stage.theme.accent; g.fillText(objectiveText(), 255, 30);
    g.fillStyle = world.remaining < 30 ? '#ff6060' : '#fff'; g.fillText('TIME ' + Math.ceil(world.remaining), 255, 51);
    g.fillStyle = '#ffe66d'; g.fillText('G ' + fmt(profile.gold) + '  ◆ ' + fmt(profile.materials), 650, 10);
    const abilities = RPG.abilityNumbers(profile);
    g.fillStyle = '#c9d5ec'; g.fillText('J ' + abilities[0].name, 650, 30);
    g.fillStyle = player.primaryCd > 0 ? '#72809a' : '#fff'; g.fillText('K ' + abilities[1].name + ' ' + (player.primaryCd > 0 ? player.primaryCd.toFixed(1) : 'READY'), 650, 46);
    g.fillStyle = player.secondaryCd > 0 ? '#72809a' : '#fff'; g.fillText('L ' + abilities[2].name + ' ' + (player.secondaryCd > 0 ? player.secondaryCd.toFixed(1) : 'READY'), 650, 62);
    g.fillStyle = '#ff9191'; g.fillText('Q HP x' + profile.potions.hp, 860, 30);
    g.fillStyle = '#8eb4ff'; g.fillText('E MP x' + profile.potions.mp, 860, 48);
    g.textBaseline = 'alphabetic';
  }

  function drawToast() {
    if (toastTimer <= 0) return;
    g.globalAlpha = clamp(toastTimer, 0, 1);
    g.font = '11px "Press Start 2P", monospace'; g.textAlign = 'center';
    const width = Math.min(820, g.measureText ? g.measureText(toastText).width + 46 : 640);
    g.fillStyle = 'rgba(5,9,18,.86)'; g.fillRect(VIEW_W / 2 - width / 2, 96, width, 34);
    g.fillStyle = toastColor; g.fillText(toastText, VIEW_W / 2, 118); g.textAlign = 'left'; g.globalAlpha = 1;
  }

  function render() {
    titleT += 0.016;
    if (state === STATE.FIELD || state === STATE.PAUSED || (state === STATE.INVENTORY && resumeState === STATE.FIELD) || state === STATE.CLEAR || state === STATE.WIN || state === STATE.OVER) {
      if (world) { drawFieldBackground(); drawWorld(); drawHUD(); }
      else drawTown();
    } else drawTown();
    drawToast();
  }

  function tick(dt) {
    if (toastTimer > 0) toastTimer -= dt;
    if (state === STATE.FIELD) updateField(dt);
  }

  function frame(now) {
    const dt = clamp((now - lastTime) / 1000, 0, 0.04);
    lastTime = now;
    tick(dt);
    render();
    if (root.requestAnimationFrame) rafId = root.requestAnimationFrame(frame);
  }

  function pause() {
    if (state === STATE.FIELD) { setState(STATE.PAUSED, 'pause'); try { Audio.musicStop(); } catch (_) {} }
    else if (state === STATE.PAUSED) { setState(STATE.FIELD, null); try { Audio.musicStart(world.stageIndex % 3); } catch (_) {} }
  }

  function handleAction(action, el) {
    Audio.ensure();
    if (action === 'new-game') { setState(STATE.CLASS, 'class'); renderClassSelect(); }
    else if (action === 'continue') continueGame();
    else if (action === 'choose-class') startNew(el.dataset.class);
    else if (action === 'start-stage') startStage(Number(el.dataset.stage));
    else if (action === 'open-shop') openShop();
    else if (action === 'open-forge') openForge();
    else if (action === 'open-skills') openSkills();
    else if (action === 'open-quests') openQuests();
    else if (action === 'open-inventory') openInventory(state);
    else if (action === 'save') saveProfile(false);
    else if (action === 'to-title') { saveProfile(true); setState(STATE.TITLE, 'title'); updateContinueButton(); }
    else if (action === 'delete-save') deleteSave();
    else if (action === 'close-panel') closeMetaPanel();
    else if (action === 'back-hub') enterHub();
    else if (action === 'buy-shop') {
      const result = RPG.buyShopEntry(profile, currentShop[Number(el.dataset.index)]);
      toast(result.ok ? '구매했습니다.' : result.reason === 'inventory-full' ? '가방이 가득 찼습니다.' : '골드가 부족합니다.', result.ok ? '#7dff9f' : '#ff7d7d');
      if (result.ok) { saveProfile(true); renderShop(); }
    }
    else if (action === 'refresh-shop') {
      const result = RPG.refreshShop(profile); toast(result.ok ? '상점 상품을 교체했습니다.' : '골드가 부족합니다.', result.ok ? '#7dff9f' : '#ff7d7d'); if (result.ok) { saveProfile(true); renderShop(); }
    }
    else if (action === 'forge-select') { forgeSelection = el.dataset.index; renderForge(); }
    else if (action === 'forge-upgrade') {
      const result = RPG.upgradeItem(profile, findForgeItem(forgeSelection));
      toast(result.ok ? '장비를 강화했습니다.' : '강화 비용 또는 조건을 확인하십시오.', result.ok ? '#ffe66d' : '#ff7d7d'); if (result.ok) { stats = RPG.deriveStats(profile); saveProfile(true); } renderForge();
    }
    else if (action === 'forge-reroll') {
      const result = RPG.rerollItem(profile, findForgeItem(forgeSelection), RPG.makeRng(Date.now()));
      toast(result.ok ? '장비 옵션을 재련했습니다.' : '재련 조건 또는 비용을 확인하십시오.', result.ok ? '#c896ff' : '#ff7d7d'); if (result.ok) saveProfile(true); renderForge();
    }
    else if (action === 'buy-skill') {
      const result = RPG.buySkill(profile, el.dataset.skill); toast(result.ok ? '스킬을 습득했습니다.' : '선행 조건 또는 포인트가 부족합니다.', result.ok ? '#ffe66d' : '#ff7d7d'); if (result.ok) { stats = RPG.deriveStats(profile); saveProfile(true); } renderSkills();
    }
    else if (action === 'claim-quest') {
      const result = RPG.claimQuest(profile, el.dataset.quest, RPG.makeRng(Date.now())); toast(result.ok ? '퀘스트 보상을 받았습니다.' : '아직 받을 수 없습니다.', result.ok ? '#ffe66d' : '#ff7d7d'); if (result.ok) saveProfile(true); renderQuests();
    }
    else if (action === 'inventory-select') { inventorySelection = Number(el.dataset.index); renderInventory(); }
    else if (action === 'equip-selected') {
      const result = RPG.equipItem(profile, inventorySelection); if (result.ok) { inventorySelection = null; stats = RPG.deriveStats(profile); saveProfile(true); toast('장비를 착용했습니다.', '#7dff9f'); } renderInventory();
    }
    else if (action === 'unequip') {
      const result = RPG.unequipItem(profile, el.dataset.slot); toast(result.ok ? '장비를 해제했습니다.' : '가방이 가득 찼습니다.', result.ok ? '#c9d5ec' : '#ff7d7d'); if (result.ok) { stats = RPG.deriveStats(profile); saveProfile(true); } renderInventory();
    }
    else if (action === 'sell-selected') {
      const result = RPG.sellItem(profile, inventorySelection); if (result.ok) { inventorySelection = null; saveProfile(true); toast('장비를 판매했습니다.', '#ffe66d'); } renderInventory();
    }
    else if (action === 'salvage-selected') {
      const result = RPG.salvageItem(profile, inventorySelection); if (result.ok) { inventorySelection = null; saveProfile(true); toast('장비를 분해했습니다.', '#92dcff'); } renderInventory();
    }
    else if (action === 'alloc-stat') {
      const result = RPG.allocateStat(profile, el.dataset.stat); if (result.ok) { stats = RPG.deriveStats(profile); saveProfile(true); } else toast('남은 능력치 포인트가 없습니다.', '#ff7d7d'); renderInventory();
    }
    else if (action === 'resume') pause();
    else if (action === 'retry-stage') startStage(world ? world.stageIndex : 0);
    else if (action === 'clear-hub' || action === 'win-hub' || action === 'over-hub') enterHub();
  }

  document.addEventListener('click', event => {
    const el = event.target && event.target.closest ? event.target.closest('[data-action]') : null;
    if (!el || el.disabled) return;
    handleAction(el.dataset.action, el);
  });
